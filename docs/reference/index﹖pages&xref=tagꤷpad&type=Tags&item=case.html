Reference - Tags - case

App: check - Page: tags/case_1
PHP
tags/case_1.php 		HTML
tags/case_1.pad 		Result 
<?php

  $language = 'FR';

?>
 
<p>
  {case $language}
    {when 'NL'}
      Hallo Wereld
    {when 'FR'}
      Bonjour le Monde
    {when 'DE'}
      Hallo Welt
    @else@
      Hello World
  {/case}
</p>

Bonjour le Monde

tags/case_2.php 		HTML
tags/case_2.pad 		Result 
<?php

  $happiness  = 50;
  $hours      = 2;

  $beer       = 10;
  $sausage    = 5;

?>
 
<p>
  {case $happiness * $hours}
    {when ($beer*2) + ($sausage*1)}
      Bonjour le Monde
    {when ($beer*4) + ($sausage*2)}
      Hallo Wereld
    {when ($beer*8) + ($sausage*4)}
      Hallo Welt
    @else@
      Hello World
  {/case}
</p>

Hallo Welt
